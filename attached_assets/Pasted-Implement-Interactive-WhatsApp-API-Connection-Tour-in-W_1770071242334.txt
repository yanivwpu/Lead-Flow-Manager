Core Driver.js Configuration (Reusable Base)Create a reusable tour component/service with these options:
javascript
import Driver from 'driver.js';
import confetti from 'canvas-confetti';

const driver = new Driver({
  allowClose: false,              // Force progression
  overlayColor: 'rgba(0,0,0,0.8)',
  stageBackground: '#ffffff',
  showProgress: true,             // Built-in progress bar
  showButtons: ['next', 'previous', 'close'],
  closeBtnText: 'Skip Tour',
  nextBtnText: 'Next →',
  prevBtnText: '← Back',
  onReset: () => { /* clear highlights */ },
  onNext: () => { /* optional validation before next */ },
});
